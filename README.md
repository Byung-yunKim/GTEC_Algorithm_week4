# GTEC_Algorithm_week4

```
class calculate():
    def __init__(self, arg1, arg2):
        self.first = arg1
        self.second = arg2
    
    def add(self):
        result = self.first + self.second
        
        return result
    
    def subtrct(self):
        result = self.first - self.second
        
        return result
    
    def multi(self):
        result = self.first * self.second
        
        return result
```
